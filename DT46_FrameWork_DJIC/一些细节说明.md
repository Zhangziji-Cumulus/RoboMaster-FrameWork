裁判系统图传串口和电管学生串口注意事项

**由于图传使用波特率为921600，DMA循环接受。电管学生串口则是115200，DMA普通模式。
使用BOARD_ID来进行区分
所以需要在usart.c里注意修改两个部分：**

void MX_USART6_UART_Init(void) 中修改添加以下内容：

/* USER CODE BEGIN USART6_Init 0 */
#if(BOARD_ID == GIMBAL_BOARD)
huart6.Init.BaudRate = 921600;
#elif(BOARD_ID == CHASSIS_BOARD)
huart6.Init.BaudRate = 115200;
#endif

void HAL_UART_MspInit(UART_HandleTypeDef* uartHandle)函数找到与USART6 DMA Init、USART6_RX Init有关的部分：修改添加以下内容：

#if(BOARD_ID == GIMBAL_BOARD)
hdma_usart6_rx.Init.Mode = DMA_CIRCULAR;
#elif(BOARD_ID == CHASSIS_BOARD)
hdma_usart6_rx.Init.Mode = DMA_NORMAL;
#endif

而stm32f4xx_it也是注意：

找到void USART6_IRQHandler(void) 添加以下内容

#if(BOARD_ID == GIMBAL_BOARD)

VT_Instance_t *inst = VT_GetInstanceByUart(&huart6);
if(inst != NULL)
{
VT_UART_IRQHandler(inst);
}

#elif(BOARD_ID == CHASSIS_BOARD)
